    init()
    t = 0
    while (t < tmax):        # MD loop
        force(f, en)         # determine forces
        integrate(f, en)     # integrate equations of motion
        t = t + delt
        sample()             # sample averages

Initialization


def init():
    for i in range(npart):
        x[i] = lattice_pos(i)    # place particles on a lattice
        v[i] = (random() - 0.5)  # give random velocities
    return

Force Calculation


def force(f, en):
    en = 0
    for i in range(npart):
        f[i] = 0                # set forces to zero
    for i in range(npart - 1):
        for j in range(i + 1, npart):  # loop over all pairs
            xr = x[i] - x[j]
            xr = xr - box * round(xr / box)  # periodic boundary conditions
            r2 = xr ** 2
            if (r2 < rc2):      # test cutoff
                r2i = 1 / r2
                r6i = r2i ** 3
                ff = 48 * r2i * r6i * (r6i - 0.5)  # Lennard-Jones potential
                f[i] = f[i] + ff * xr              # update forces
                f[j] = f[j] - ff * xr              #
                en = en + 4 * r6i * (r6i - 1) - ecut  # update energy
    return
